private void initializePopulation(double minXi, double maxXi)
 {
     string individu = "";
     string gene;
 
     for (int i = 0; i < this.popSize[0] ; ++i)
     {
         for (int g = 0; g < this.dimension; ++g)
         {
             if (g > 0)
                 individu += "|";
             gene = Tools.decToBinProportional(random.NextDouble() * (maxXi - minXi) + minXi, minXi, maxXi);
             while (gene.Length < Tools.nb_bit)
                 gene = "0" + gene;
             individu += gene;
         }
         population[i] = individu;
         individu = "";
     }
 }
