public List<string> selection(int pSelectionMethod)

     List<string> newPop = new List<string>();
 
     #region Selection Method 1 : The wheel
     if (pSelectionMethod == 1)
     {   
         var mySum = this.ratio.Skip(0).Take((int)popSize[0]).Sum();
         int[] wheel = new int[mySum];
         int iWheel = 0;
 
         /// Creation of the wheel
         /// With proportional value for every rank
         for (int i = 1; i <= popSize[0]; ++i)
         {
             for (int j = 0; j < i; ++j )
             {
                 wheel[iWheel] = ratio[i - 1];
                 iWheel++;
             }
         }
 
         /// The draw with the wheel, select induviduals
         for (int i = 0; i < popSize[0] / 2; ++i)
         {
             int tirage = this.random.Next(mySum);
             int winningRatio = wheel[tirage];
             newPop.Add(population[winningRatio - 1]);
             Console.WriteLine("Wheel[{0}] = {1} => {2}\nInduvidual : {3}", tirage, winningRatio, results[winningRatio - 1], population[winningRatio - 1]);
         }
 
         return newPop;
     }
